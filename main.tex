		class AlDeSCo {
			public int magic(int |\tikzmark{n}|n, boolean b) {
				int x = 42;
				if (b)
					x = addFive(n);
				else
					x = n;
				return x;
			}
			private int addFive(int m) {
				return m + 5;
			}
		}
	\end{minted}
	\begin{tikzpicture}[overlay, remember picture]
		\node[anchor=east, text width=0.35\textwidth, xshift=-1cm] at (current page.east) {\block%
			How does \subnode{t}{\mintinline{java}{n}} propagate through the programm?%
		\endblock};
